long long cal(vector<int>& arr,int h)
{
    long long totalhours=0;
    for(int i=0;i<arr.size();i++)
    {
        totalhours += ((long long)arr[i] + h - 1) / h;
    }
    return totalhours;
}

class Solution {
public:
    int minEatingSpeed(vector<int>& piles, int h) {
        sort(piles.begin(),piles.end());
        int n=piles.size();
        int low=1,high=piles[n-1];
        while(low<=high)
        {
           int mid=low+(high-low)/2;
            long long  total=cal(piles,mid);
            if(total<=h)
            {
                high=mid-1;
            }
            else {
                low=mid+1;
            }

        }        
        return low;
        
    }
};
