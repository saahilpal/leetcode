**Pattern:** Binary Search  
**Language:** cpp

## Approach
first i'll copy whole one vector in one then i'll use stl sort  after sorting  i'll find the median by applying the formula  if the size of array is even i'll add two middle element then i'll the take the half

## Code

```cpp
class Solution {
public:
    double findMedianSortedArrays(vector<int>& nums1, vector<int>& nums2) {
        nums1.insert(nums1.end(), nums2.begin(), nums2.end());
        sort(nums1.begin(), nums1.end());
        int n = nums1.size();
        double res;

        if (n % 2 == 0) {
            int mid = n / 2;
            int mid1 = n / 2 - 1;
            res = (nums1[mid] + nums1[mid1]) / 2.0;
        } else {
            int mid = (n / 2);
            res = nums1[mid] ;
        }

        return res;
    }
};
```
