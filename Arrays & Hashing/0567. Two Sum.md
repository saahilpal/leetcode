class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int, int> mp;
        vector<int> res;
        for (int i = 0; i < nums.size(); i++) {
            int need = target - nums[i];
            if (mp.find(need) != mp.end()) {
                int j=mp[need];
                res.push_back(i);
                res.push_back(j);
                break;
            }
            mp[nums[i]] = i;
        }
        return res;
    }
};
